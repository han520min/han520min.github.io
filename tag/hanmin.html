# 加密方法
def make_password(mypass):
    # md5算法
    mypass = hashlib.md5()
    # 对字符串进行编译(加盐操作)
    mypass.update('123'.encode(encoding='utf-8'))
    # 返回密文
    return mypass.hexdigest()
